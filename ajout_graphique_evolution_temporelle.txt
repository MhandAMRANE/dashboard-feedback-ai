import pandas as pd
from flask import render_template

def compute_weekly_sentiment(feedbacks):
    df = pd.DataFrame(feedbacks)
    df['date'] = pd.to_datetime(df['feedback_date_raw'])
    df['week'] = df['date'].dt.to_period('W').astype(str)
    # Compter les sentiments par semaine
    weekly_counts = df.groupby('week')['sentiment'].value_counts().unstack(fill_value=0).reset_index()
    labels = weekly_counts['week'].tolist()
    data_positifs = weekly_counts.get('positif', pd.Series([0]*len(labels))).tolist()
    data_neutres = weekly_counts.get('neutre', pd.Series([0]*len(labels))).tolist()
    data_negatifs = weekly_counts.get('negatif', pd.Series([0]*len(labels))).tolist()
    return labels, data_positifs, data_neutres, data_negatifs
